```cpp
#include <bits/stdc++.h>
#include <limits>
#define all(x) begin(x),end(x)
#define fir first
#define sec second
 
using ll = long long;
using namespace std;

void setIO(string s){
	freopen((s + ".in").c_str(),"r",stdin);
	freopen((s + ".out").c_str(),"w",stdout);
	}



void solve(){
    int t;
    cin >> t;
    while(t--){
        int n;
        cin >> n;
        vector<ll> m;
        vector<ll> d;
        for(int i{}; i <( 2 * n); i++){
            int x,y;
            cin >> x >> y;
            if (!x) m.push_back(abs(y));
            else if (!y) d.push_back(abs(x));
        }

        sort(all(m));
        sort(all(d));

        double ans = 0;
        for(int i{}; i < n; i++){
            ans += sqrt((double)((m[i]) * (m[i])) + ((d[i]) * (d[i])));
        }
        cout << setprecision(10) << ans << '\n';
    }
}

int main(){
    //setIO("t");
	ios_base::sync_with_stdio(false);
	cin.tie(nullptr);
	solve();
	return 0;
}
```
