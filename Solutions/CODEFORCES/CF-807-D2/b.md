void solve(){
    int n;
    cin >> n;
 
    vi a(n);
    for(auto& i : a) cin >> i;
 
 
    ll sm = 0;
    //max(number of 0s, n-number of 0s)    
    int cur = -1;
    for(int i{}; i < n-1; i++){
        if (!a[i] && cur == -1) continue;
        cur = 1;
        if (!a[i] && cur){
            sm++;
        }
        sm += a[i];
    }
 
    cout << sm << '\n';
}
 
int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    int t = 1;
    cin >> t;
    while(t--){
        solve();
    }
	return 0;
}
