void solve(){
    int n;
    cin >> n;
 
    vi a(5 * n);
    for(auto& i : a) cin >> i;
 
    sort(all(a));
 
    int sm = 0;
    for(int i{n}; i < 4*n; i++){
        sm += a[i];
    }
    cout << setprecision(5) << fixed;
    cout << (double)sm/(3*n);
 
}

    while(t--){
        solve();
    }

}
