#include <bits/stdc++.h>
#include <limits>
#define all(x) begin(x),end(x)
 
 
using ll = long long;
using namespace std;
 
void setIO(string s){
	freopen((s + ".in").c_str(),"r",stdin);
	freopen((s + ".out").c_str(),"w",stdout);
	}
 
void solve(){
	int n;
	cin >> n;
	vector<int> a(n);
	for(auto& i: a) cin >> i;
	sort(all(a));
	for(auto& i: a) cout << i << ' ';
}
 
int main(){
	ios_base::sync_with_stdio(false);
	cin.tie(nullptr);
	solve();
 
	return 0;
	}
