void solve(){
    int n;
    cin >> n;
    int top = n;
    set<int> s;
    for(int i{}; i < n; i++){
        int x;
        cin >> x;
        s.insert(x);
        if (x == top){
            while (s.count(x)){
                cout << x << ' ';
                x--;
                top--;
            }
        }
        cout << '\n';
    }
        
}

int main(){
	ios_base::sync_with_stdio(false);
	cin.tie(nullptr);
	solve();

}
