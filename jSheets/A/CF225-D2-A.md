void solve(){
    int n;
    cin >> n;
    int top;
    cin >> top;
    vector<pair<int,int>> faces(n);
    for(auto& i :faces) cin >> i.fir >> i.sec;

    for(auto& i : faces){
        set<int> facen ={1,2,3,4,5,6};
        facen.erase(i.fir);
        facen.erase(7-i.fir);
        facen.erase(i.sec);
        facen.erase(7-i.sec);
        if (facen.count(top)){
            facen.erase(top);
            top = *facen.begin();
        }else{
            cout << "NO";
            return;
        }
    }
    cout << "YES";
}


int main(){
    //setIO("t");
	ios_base::sync_with_stdio(false);
	cin.tie(nullptr);
	solve();
	return 0;
}
