#include <bits/stdc++.h>

void setIO(string s){
	freopen((s + ".in").c_str(),"r",stdin);

void solve(){
    int n;
    cin >> n;
    vector<ll> a(n);
    for(auto& i : a) cin >> i;

    int q;
    cin >> q;
    vector<ll> h(q);
    for(auto& i : h) cin >> i;

    for(auto& i : h){
        int lo = 0, hi = n-1;

        while (hi > lo){
            int mid = lo + (hi-lo+1)/2;
            if (a[mid] < i){
                lo = mid;
            }else{
                hi = mid-1;
            }
        }

        if (a[lo] < i) cout << a[lo];
        else cout << "X";
        cout << ' ';

        lo = 0, hi = n-1;
        while (hi > lo){
            int mid = lo + (hi-lo)/2;
            if (a[mid] > i){
                hi = mid;
            }else{
                lo = mid+1;
            }
        }
        if (a[hi] > i) cout << a[hi];
        else cout << "X";
        cout << '\n';
    }
}

int main(){
    //setIO("t");
	ios_base::sync_with_stdio(false);
	cin.tie(nullptr);
	solve();
	return 0;
}
