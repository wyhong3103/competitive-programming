Testing bipartiteness using DFS

```cpp
#include <bits/stdc++.h>
#include <limits>
#define all(x) begin(x),end(x)
#define fir first
#define sec second
 
using ll = long long;
using namespace std;
 
void setIO(string s){
	freopen((s + ".in").c_str(),"r",stdin);
	freopen((s + ".out").c_str(),"w",stdout);
	}
 
bool vis[101];
vector<int> adj[101];
bool color[101];
 
 
bool dfs(int node){
    vis[node] = true;
 
    for(auto& i : adj[node]){
        if (!vis[i]){
            vis[i] = true;
            color[i] = !color[node];
            if (!dfs(i)){
                return false;
            }
        }
        else if (color[i] == color[node]){
            return false;
        }
    }
    return true;
}
 
 
void solve(){
    int n,m;
    cin >> n >> m;
 
    for(int i{}; i < m; i++){
        int e1,e2;
        cin >> e1 >> e2;
        e1--;e2--;
        adj[e1].push_back(e2);
        adj[e2].push_back(e1);
    }
 
    
    int bi{};
    for(int i{}; i < n; i++){
        if (!vis[i]){
            if (!dfs(i)) bi++;
        }
    }
    
    cout << bi + ((n-bi) % 2 == 0 ? 0 : 1);
}
 
int main(){
	//ios_base::sync_with_stdio(false);
	//cin.tie(nullptr);
	solve();
	return 0;
}
```
