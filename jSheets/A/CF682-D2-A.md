void solve(){

    cin >> n >> m;
    
    ll ans = 0;

    for(int i{1}; i <= n; i++){
        bool found = false;
        int start = 0;
        if (!found){
            for(int j{1}; j <= m; j++){
                if ((i + j) % 5 == 0){
                    start = j;
                    found = true;
                    break;
                }
            }
        }
        if (found){
            ans +=ceil((double)(m-(start-1))/5);
        }
    }
    cout << ans;
}

    int n,m;

        a[i%5]++;
    }

    for(int i{1}; i <= m; i++){
        b[i%5]++;
    }

    ll ans = (a[0] * b[0]) + (a[1] * b[4]) + (a[2] * b[3]) + (a[3] * b[2]) + (a[4] * b[1]);
    cout << ans;
}
