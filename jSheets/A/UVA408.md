int gcd(int a,int b){
    if (b == 0) return a;
    else return gcd(b, a % b);
}

void solve(){
    int step, mod;

    while(cin >> step >> mod){
        cout << setw(10) << step << setw(10) << mod << "    " << (gcd(step,mod) == 1 ? "Good Choice" : "Bad Choice") << "\n\n";
    }
}

int main(){

	cin.tie(nullptr);
	solve();

}
