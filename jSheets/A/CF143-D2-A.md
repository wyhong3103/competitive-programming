void solve(){
    int r1,r2,c1,c2,d1,d2;
    cin >> r1 >> r2 >> c1 >> c2 >> d1 >> d2;

    for(int i{1}; i < 10; i++){
        set<int> used;
        used.insert(i);
        for(int j{1}; j < 10; j++){
            if (used.count(j)) continue;
            used.insert(j);
            for(int k{1}; k < 10; k++){
                if (used.count(k)) continue;
                used.insert(k);
                for(int l{1}; l < 10; l++){
                    if (used.count(l)) continue;
                    if (i + j == r1 && k + l == r2 && i + l == d1 && j + k == d2 && i + k == c1 && l + j == c2){
                        cout << i << ' ' << j << '\n' << k << ' ' << l << '\n';
                        return;
                    }
                }
                used.erase(k);
            }
            used.erase(j);
        }
    } 
    cout << -1;
}


int main(){
