    int n;

        if (i == 0){
            cout << abs(a[i]- a[i+1]) << ' ' <<abs(a[i]-a[n-1]) << '\n';

            cout << abs(a[i] - a[i-1]) << ' ' << abs(a[i] - a[0]) << '\n';

        }
    }
}
