ll rePow(ll base, ll power){
    ll ans = 1;
    for(ll i{1}; i <= power; i++){
        ans *= base;
    }
    return ans;
}

void solve(){
    ll n; int k; 
    cin >> n >> k;
    
    ll lo = 0, hi = n;
    while (hi > lo){
        ll mid = lo + (hi-lo)/2;
        
        ll tk = 1;
        ll sm = mid;
        while ((mid/rePow(k,tk)) != 0){
            sm += (mid/ rePow(k,tk));
            tk++;
        }
        if (sm >= n){
            hi = mid;
        }else{
            lo = mid+1;
        }
    }
    cout << lo;
}

int main(){

	solve();
